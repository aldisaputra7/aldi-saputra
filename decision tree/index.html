    # menentukan nilai entropy target

def entropy(T):
    hasil=0
    jumlah=0
    for y in T:
        jumlah+=y
    for z in range (len(T)):
        if jumlah!=0:
            T[z]=T[z]/jumlah

    for i in T:
        if i != 0:
            hasil-=i*math.log(i,2)
    return hasil
